# 异步抠图处理机制改进计划

## 问题分析

### 当前问题
1. **主线程阻塞**：`handleBatchSetMatting` 使用同步 for 循环，每次调用 `removeImageBackground` 都会阻塞主线程
2. **无进度反馈**：仅有简单的文字提示，缺少进度条和百分比显示
3. **无法取消**：处理过程中用户无法终止任务
4. **无法并发**：串行处理效率低下
5. **界面卡顿**：AI 背景移除操作耗时较长，导致整个页面无响应

### 根本原因
- `@imgly/background-removal` 库的 `removeBackground` 函数是 CPU 密集型操作
- 当前实现在主线程中同步执行，阻塞了 React 的渲染和事件处理

## 解决方案

### 核心架构：Web Worker + 任务队列

使用 Web Worker 将抠图任务移至后台线程执行，同时实现任务队列管理系统。

```

│  │   组件      │◄─┤   (React)   │◄─┤ (ProgressDisplay)   │  │
```

### 1. Web Worker 实现 (mattingWorker.ts)

```typescript
// emo-j/src/workers/mattingWorker.ts
import { removeBackground } from '@imgly/background-removal';

interface WorkerMessage {
  type: 'process' | 'cancel';
  taskId: string;
  blob?: Blob;
}

interface WorkerResponse {
  taskId: string;
  status: 'processing' | 'completed' | 'failed' | 'cancelled';
  progress?: number;
  result?: Blob;
  error?: string;
}

let currentAbortController: AbortController | null = null;

self.onmessage = async (e: MessageEvent<WorkerMessage>) => {
  const { type, taskId, blob } = e.data;

  if (type === 'cancel') {
    currentAbortController?.abort();
    return;
  }

  if (type === 'process' && blob) {
    currentAbortController = new AbortController();
    
    try {
      self.postMessage({ taskId, status: 'processing', progress: 0 });
      
      const result = await removeBackground(blob, {
        signal: currentAbortController.signal,
        progress: (key, current, total) => {
          const progress = Math.round((current / total) * 100);
          self.postMessage({ taskId, status: 'processing', progress });
        }
      });
      
      self.postMessage({ taskId, status: 'completed', result });
    } catch (err) {
      if (err.name === 'AbortError') {
        self.postMessage({ taskId, status: 'cancelled' });
      } else {
        self.postMessage({ taskId, status: 'failed', error: err.message });
      }
    }
  }
};

export {};
```

### 2. 任务队列管理器 (taskQueueManager.ts)

```typescript
// emo-j/src/utils/taskQueueManager.ts
export type TaskStatus = 'pending' | 'processing' | 'completed' | 'failed' | 'cancelled';

export interface MattingTask {
  id: string;
  blob: Blob;
  index: number;
  status: TaskStatus;
  progress: number;
  result?: Blob;
  error?: string;
  startTime?: number;
  endTime?: number;
}

export type TaskEventType = 'add' | 'start' | 'progress' | 'complete' | 'fail' | 'cancel';
export type TaskEventCallback = (task: MattingTask, event: TaskEventType) => void;

export class TaskQueueManager {
  private queue: MattingTask[] = [];
  private workers: Worker[] = [];
  private maxConcurrent: number;
  private eventListeners: Map<TaskEventType, Set<TaskEventCallback>> = new Map();
  private taskWorkerMap: Map<string, Worker> = new Map();
  
  constructor(maxConcurrent = 2) {
    this.maxConcurrent = maxConcurrent;
    this.initWorkers();
  }
  
  private initWorkers(): void {
    for (let i = 0; i < this.maxConcurrent; i++) {
      const worker = new Worker(
        new URL('../workers/mattingWorker.ts', import.meta.url),
        { type: 'module' }
      );
      worker.onmessage = this.handleWorkerMessage.bind(this);
      this.workers.push(worker);
    }
  }
  
  addTask(blob: Blob, index: number): string {
    const task: MattingTask = {
      id: `task_${Date.now()}_${index}`,
      blob,
      index,
      status: 'pending',
      progress: 0
    };
    this.queue.push(task);
    this.emit(task, 'add');
    this.processQueue();
    return task.id;
  }
  
  addTasks(tasks: Array<{ blob: Blob; index: number }>): string[] {
    return tasks.map(t => this.addTask(t.blob, t.index));
  }
  
  cancelTask(taskId: string): void {
    const task = this.queue.find(t => t.id === taskId);
    if (task && task.status === 'processing') {
      const worker = this.taskWorkerMap.get(taskId);
      worker?.postMessage({ type: 'cancel', taskId });
    } else if (task) {
      task.status = 'cancelled';
      this.emit(task, 'cancel');
    }
  }
  
  cancelAll(): void {
    this.queue.forEach(task => {
      if (task.status === 'pending' || task.status === 'processing') {
        this.cancelTask(task.id);
      }
    });
  }
  
  getProgress(): { total: number; completed: number; percentage: number; processing: number } {
    const total = this.queue.length;
    const completed = this.queue.filter(t => t.status === 'completed').length;
    const processing = this.queue.filter(t => t.status === 'processing').length;
    const percentage = total > 0 ? Math.round((completed / total) * 100) : 0;
    return { total, completed, percentage, processing };
  }
  
  getTask(taskId: string): MattingTask | undefined {
    return this.queue.find(t => t.id === taskId);
  }
  
  getAllTasks(): MattingTask[] {
    return [...this.queue];
  }
  
  clearCompleted(): void {
    this.queue = this.queue.filter(t => t.status !== 'completed' && t.status !== 'cancelled');
  }
  
  on(event: TaskEventType, callback: TaskEventCallback): () => void {
    if (!this.eventListeners.has(event)) {
      this.eventListeners.set(event, new Set());
    }
    this.eventListeners.get(event)!.add(callback);
    return () => this.off(event, callback);
  }
  
  off(event: TaskEventType, callback: TaskEventCallback): void {
    this.eventListeners.get(event)?.delete(callback);
  }
  
  private emit(task: MattingTask, event: TaskEventType): void {
    this.eventListeners.get(event)?.forEach(cb => cb(task, event));
  }
  
  private processQueue(): void {
    const pendingTasks = this.queue.filter(t => t.status === 'pending');
    const idleWorkers = this.workers.filter(w => 
      !Array.from(this.taskWorkerMap.values()).includes(w)
    );
    
    while (pendingTasks.length > 0 && idleWorkers.length > 0) {
      const task = pendingTasks.shift()!;
      const worker = idleWorkers.shift()!;
      
      task.status = 'processing';
      task.startTime = Date.now();
      this.taskWorkerMap.set(task.id, worker);
      this.emit(task, 'start');
      
      worker.postMessage({
        type: 'process',
        taskId: task.id,
        blob: task.blob
      });
    }
  }
  
  private handleWorkerMessage(e: MessageEvent): void {
    const { taskId, status, progress, result, error } = e.data;
    const task = this.queue.find(t => t.id === taskId);
    
    if (!task) return;
    
    if (status === 'processing') {
      task.progress = progress ?? task.progress;
      this.emit(task, 'progress');
    } else if (status === 'completed') {
      task.status = 'completed';
      task.result = result;
      task.endTime = Date.now();
      this.taskWorkerMap.delete(taskId);
      this.emit(task, 'complete');
      this.processQueue();
    } else if (status === 'failed') {
      task.status = 'failed';
      task.error = error;
      task.endTime = Date.now();
      this.taskWorkerMap.delete(taskId);
      this.emit(task, 'fail');
      this.processQueue();
    } else if (status === 'cancelled') {
      task.status = 'cancelled';
      task.endTime = Date.now();
      this.taskWorkerMap.delete(taskId);
      this.emit(task, 'cancel');
      this.processQueue();
    }
  }
  
  destroy(): void {
    this.workers.forEach(w => w.terminate());
    this.workers = [];
    this.queue = [];
    this.taskWorkerMap.clear();
  }
}
```

### 3. 进度显示组件 (ProgressDisplay.tsx)

```tsx
// emo-j/src/components/ProgressDisplay/ProgressDisplay.tsx
import { useEffect, useState } from 'react';
import './ProgressDisplay.css';

export interface ProgressDisplayProps {
  total: number;
  completed: number;
  processing: number;
  currentTaskIndex: number | null;
  isProcessing: boolean;
  onCancel: () => void;
  onComplete?: () => void;
}

export function ProgressDisplay({
  total,
  completed,
  processing,
  currentTaskIndex,
  isProcessing,
  onCancel,
  onComplete
}: ProgressDisplayProps) {
  const [showNotification, setShowNotification] = useState(false);
  const percentage = total > 0 ? Math.round((completed / total) * 100) : 0;
  
  useEffect(() => {
    if (completed === total && total > 0 && isProcessing === false) {
      setShowNotification(true);
      onComplete?.();
      const timer = setTimeout(() => setShowNotification(false), 3000);
      return () => clearTimeout(timer);
    }
  }, [completed, total, isProcessing, onComplete]);
  
  if (!isProcessing && completed === 0) return null;
  
  return (
    <div className="progress-display">
      <div className="progress-header">
        <span className="progress-title">
          {isProcessing ? '正在处理抠图...' : '处理完成'}
        </span>
        <span className="progress-stats">
          {completed} / {total} ({percentage}%)
        </span>
      </div>
      
      <div className="progress-bar-container">
        <div 
          className="progress-bar-fill"
          style={{ width: `${percentage}%` }}
        />
      </div>
      
      {isProcessing && currentTaskIndex !== null && (
        <div className="progress-current">
          正在处理第 {currentTaskIndex + 1} 张图片...
          {processing > 1 && ` (同时处理 ${processing} 张)`}
        </div>
      )}
      
      <div className="progress-actions">
        {isProcessing && (
          <button className="progress-cancel-button" onClick={onCancel}>
            取消处理
          </button>
        )}
      </div>
      
      {showNotification && (
        <div className="progress-notification">
          ✓ 所有抠图任务已完成！
        </div>
      )}
    </div>
  );
}
```

```typescript
// emo-j/src/utils/notifications.ts
export async function requestNotificationPermission(): Promise<boolean> {
  if (!('Notification' in window)) return false;
  
  if (Notification.permission === 'granted') return true;
  
  if (Notification.permission !== 'denied') {
    const permission = await Notification.requestPermission();
    return permission === 'granted';
  }
  
  return false;
}

export function showNotification(title: string, options?: NotificationOptions): void {
  if (Notification.permission === 'granted') {
    new Notification(title, {
      icon: '/favicon.ico',
      badge: '/favicon.ico',
      ...options
    });
  }
}

export function notifyMattingComplete(count: number): void {
  showNotification('抠图完成', {
    body: `成功处理 ${count} 张图片`,
    tag: 'matting-complete'
  });
}

export function notifyMattingFailed(count: number): void {
  showNotification('抠图失败', {

  });
```

### 风险1：Web Worker 兼容性

- **缓解**：限制并发数量，及时清理完成的任务，释放 Blob 引用

### 风险3：大图片处理
- **缓解**：添加图片大小限制，提供压缩选项
